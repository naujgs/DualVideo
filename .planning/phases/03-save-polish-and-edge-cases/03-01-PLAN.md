**Step 2 — Implement PhotoSaveManager (GREEN).**

Create `DualVideo/Features/Recording/PhotoSaveManager.swift`:
```swift
import Photos
import Foundation
import os.log

private let logger = Logger(subsystem: "com.naujgs.DualVideo", category: "PhotoSaveManager")

enum PhotoSaveError: Error, Equatable {
    case permissionDenied
    case saveFailed(String)  // use error.localizedDescription; Equatable-safe string
}

final class PhotoSaveManager {
    // Testable injection point: defaults to real PHPhotoLibrary status
    private let statusProvider: () -> PHAuthorizationStatus
    // Testable injection point: defaults to real PHPhotoLibrary.shared()
    private let performChanges: (@escaping () -> Void, @escaping (Bool, Error?) -> Void) -> Void

    init(
        statusProvider: @escaping () -> PHAuthorizationStatus = {
            PHPhotoLibrary.authorizationStatus(for: .addOnly)
        },
        performChanges: @escaping (@escaping () -> Void, @escaping (Bool, Error?) -> Void) -> Void = {
            PHPhotoLibrary.shared().performChanges($0, completionHandler: $1)
        }
    ) {
        self.statusProvider = statusProvider
        self.performChanges = performChanges
    }

    /// Save video file at url to Photos. Deletes temp file only on success.
    /// Calls completion on the MAIN thread.
    func saveVideoToPhotos(url: URL, completion: @escaping (Result<Void, PhotoSaveError>) -> Void) {
        // DEV-03: re-check authorization status at save time (user may have revoked in Settings)
        let status = statusProvider()
        guard status == .authorized || status == .limited else {
            DispatchQueue.main.async { completion(.failure(.permissionDenied)) }
            return
        }

        // OUT-01: save via PhotoKit — the only approved API for .addOnly permission
        performChanges({
            PHAssetChangeRequest.creationRequestForAssetFromVideo(atFileURL: url)
        }, { success, error in
            // Always dispatch to main before updating @Observable state (Pitfall 4)
            DispatchQueue.main.async {
                if success {
                    // OUT-01: temp file deleted only after confirmed Photos save (Pitfall 5)
                    try? FileManager.default.removeItem(at: url)
                    logger.info("PhotoSaveManager: saved and cleaned up \(url.lastPathComponent)")
                    completion(.success(()))
                } else {
                    let msg = error?.localizedDescription ?? "unknown"
                    logger.error("PhotoSaveManager: save failed — \(msg)")
                    completion(.failure(.saveFailed(msg)))
                }
            }
        })
    }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| app → PHPhotoLibrary | User-granted .addOnly permission controls write access to Camera Roll |
| performChanges completion → @Observable state | Background-queue completion updates main-thread state |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01-01 | Elevation of Privilege | PHPhotoLibrary.authorizationStatus re-check | mitigate | `authorizationStatus(for: .addOnly)` called immediately before `performChanges`; failure returns `permissionDenied` and shows Settings alert |
| T-03-01-02 | Information Disclosure | temp .mov preserved on save failure | accept | File is in app tmp dir, inaccessible to other apps; `cleanUpOrphanedTempFiles` removes it on next launch; low-value data |
| T-03-01-03 | Denial of Service | performChanges completion on background queue mutates @Observable | mitigate | All state updates inside `DispatchQueue.main.async` in completion handler; validated by Swift 6 strict concurrency |
</threat_model>
